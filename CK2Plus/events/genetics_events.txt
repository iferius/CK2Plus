character_event = {
	id = 1000182974

	hide_window = yes

		NOT = { age = 1 }
		NOT = { has_character_flag = genetics_check }

		hidden_tooltip = {

					limit = {
						OR = {
							mother = { trait = genius }
							mother = { trait = quick }
						}
						OR = {
							father_even_if_dead = { trait = genius }
							father_even_if_dead = { trait = quick }
						}
						NOT = {
							mother = { trait = imbecile }
							mother = { trait = slow }
							father_even_if_dead = { trait = imbecile}
							father_even_if_dead = { trait = slow }
						}
					}
					random_list = {
						75 = { }
						20 = { add_trait = quick }
						5 = { add_trait = genius }
					}
				}
				if = {
					limit = {
						OR = {
							mother = { trait = genius }

			#chance for slow or imbecile
			if = {
				limit = {

					NOT = { trait = slow }
					NOT = { trait = imbecile }

#Reduce fertility / change stats for older characters
character_event = {
	id = 1000182975
	picture = "GFX_evt_illness"

	hide_from = yes
	hide_window = yes

	trigger = {
		NOT = { trait = immortal }
		age = 50
		NOT = { age = 60 }
		OR = {
			trait = young1
			trait = young6
			trait = young16
			NOT = { trait = old50 }
			trait = old60
			trait = old70
			trait = old80
			trait = old90
		}
	}
	
	immediate = {
		remove_trait = young1
		remove_trait = young6
		remove_trait = young16
		add_trait = old50
		remove_trait = old60
		remove_trait = old70
		remove_trait = old80
		remove_trait = old90
	}
	
	mean_time_to_happen = { days = 1 }

	desc = "AI_EVENT"
}
