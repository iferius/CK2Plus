	id = 55003
	desc = "EVTDESC55003"
	picture = GFX_evt_throne_room

	trigger = {
		year = 1100
		any_demesne_province = {
			OR = {
				kingdom = {
					title = k_england
				} 
				kingdom = {
					title = k_englandnorman
				}
				kingdom = {
					title = k_englandnorwegian
				}
				empire = {
					title = e_britannia
				}
				empire = {
					title = e_england
				}
			}
			OR = {
				culture = saxon
				culture = norman
				culture = frankish
				culture = occitan
			}
			OR = {
				AND = {
					any_neighbor_province = { has_owner = yes culture = english }

				}

					culture = english

						independent = yes
