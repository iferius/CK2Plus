#Mingling event
#Unreformed culture provinces converting
province_event = {
	id = 1000191800
	desc = EVTDESC1000191800
	trigger = {
		has_global_flag = another_new_dawn
		AND = {
		OR = {
		culture = tribal
		culture = germanic
		culture = slavic
		culture = italic
		culture = celtic
		culture = nordic
		culture = baltic
		culture = ugric
		culture = altaic
		culture = kartvelian
		culture = hellenic
		culture = iranian
		culture = hamitic
		culture = west_african
		}
		AND = {
		NOT = {owner = { culture = tribal}}
		NOT = {owner = { culture = germanic}}
		NOT = {owner = { culture = slavic}}
		NOT = {owner = { culture = italic}}
		NOT = {owner = { culture = celtic}}
		NOT = {owner = { culture = nordic}}
		NOT = {owner = { culture = baltic}}
		NOT = {owner = { culture = ugric}}
		NOT = {owner = { culture = altaic}}
		NOT = {owner = { culture = kartvelian}}
		NOT = {owner = { culture = hellenic}}
		NOT = {owner = { culture = iberian}}
		NOT = {owner = { culture = iranian}}
		NOT = {owner = { culture = semitic}}
		NOT = {owner = { culture = hamitic}}
		NOT = {owner = { culture = west_african}}
		}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		owner = {
			reverse_culture = ROOT
		}
	}
	option = {
		name = PERFECT
	}
}
