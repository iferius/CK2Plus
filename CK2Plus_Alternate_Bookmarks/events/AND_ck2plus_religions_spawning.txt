	desc = EVTDESC1000190001

				remove_title = job_steward
				remove_title = job_spymaster

			}
		}
		any_realm_province = {

#						religion = FROM
#					}
#				}
#			}
			religion = FROM
			any_province_character = { religion = FROM }
		}
		any_realm_province = { # Prophet location gets holy city event
			limit = {
				any_province_character = { # prophets location
					AND = {
						has_character_modifier = prophet
						religion = FROM
					}
				}
			}
			province_event = { id = 1000190004 }
		}
		IF = {
			limit = {
				independent = no
			}
			liege = {
				character_event = { id = 1000190002 }
			}
		}
	}
	option = {
		name = EVTOPTB1000190001
		trigger = {
			ai = no
		}
		FROM = {
			banish = yes

	id = 1000190005
	desc = EVTDESC1000190005
	picture = "GFX_evt_stone_church"
	
	trigger = {
		has_province_flag = holy_region
		ruler = { religion = ROOT }
		NOT = { has_province_flag = holy_structure }
		NOT = { has_province_flag = holy_structure_constr }
	}

	mean_time_to_happen = {
		days = 3650
	}
	
	immediate = {
		set_province_flag = holy_structure_constr
	}
	
	option = { # Give resources
		name = EVTOPTA1000190005
		trigger = {
			wealth = 100
		}
		wealth = -100
		hidden_tooltip = {
			ruler = {
				character_event = { id = 1000190006 days = 365 }

		}
	}
	option = { # Send men
		name = EVTOPTB1000190005
		add_province_modifier = { name = reduced_levies duration = 365 }

				character_event = { id = 1000190006 days = 365 }
			}
		}

# Prophet appears
province_event = {
	id = 1000190009
	hide_window = yes
	hide_from = yes
	
	trigger = {
		has_global_flag = another_new_dawn
		is_land = yes
		ruler = {
			has_character_flag = prophetspawn
		}
	}
	
	immediate = {
		ruler = { clr_character_flag = prophetspawn }
		IF = {
			limit = {
				has_global_flag = ancestral_founded
				has_global_flag = astrolatry_founded
				has_global_flag = pantheism_founded
			}
			set_global_flag = basic_religions_founded
			clr_global_flag = ancestral_founded
			clr_global_flag = astrolatry_founded
			clr_global_flag = pantheism_founded
		}
		IF = {
			limit = {
				NOT = {
					top_liege = {
						any_realm_character = {
							has_character_modifier = prophet
						}
					}
				}
				ruler = { religion = folklore }
			}
			random_list = {	# random allocation of religions. The follow-up events can result in failure if the religion already has been founded
				33 = {	# Astrolatry
					random_province_character = {
						limit = {
							ai = yes
							OR = {
								trait = detached_priest
								trait = martial_cleric
								trait = scholarly_theologian
								trait = mastermind_theologian
								trait = scholar
							}
						}
						character_event = { id = 1000190020 }
					}
				}
				34 = {	# Pantheistic
					random_province_character = {
						limit = {
							ai = yes
							OR = {
								trait = detached_priest
								trait = martial_cleric
								trait = scholarly_theologian
								trait = mastermind_theologian
								trait = scholar
							}
						}
						character_event = { id = 1000190040 }
					}
				}
				33 = {	# Ancestral
					random_province_character = {
						limit = {
							ai = yes
							OR = {
								trait = detached_priest
								trait = martial_cleric
								trait = scholarly_theologian
								trait = mastermind_theologian
								trait = scholar
							}
						}
						character_event = { id = 1000190060 }
					}
				}
			}
		}
		IF = {
			limit = {
				AND = {
					NOT = {
						any_realm_character = {
							ai = yes
							OR = {
								trait = detached_priest
								trait = martial_cleric
								trait = scholarly_theologian
								trait = mastermind_theologian
								trait = scholar
							}
						}
					}
					NOT = {
						top_liege = {
							any_realm_character = {
								has_character_modifier = prophet
							}
						}
					}
					ruler = { religion = folklore }
				}
			}
			ROOT = { province_event = { id = 1000190010 days = 10 } }
		}
	}
}

						}

						}
					}
				}
			}
		}
	}
}

# Scope to ill/possessed ruler of a poor province
character_event = {
	id = 1000190080
	hide_window = yes
	hide_from = yes
	
	only_rulers = yes
	min_age = 25
	trigger = {
		has_global_flag = another_new_dawn
		OR = { # there must be a not founded religion
			NOT = { has_global_flag = polytheism_founded }
			NOT = { has_global_flag = monotheism_founded }
		}
		tier = DUKE
		any_realm_province = { # a poorly performing province must exist in the realm
			OR = {
				has_province_modifier = thieves_guild
				has_province_modifier = smugglers_ring
				has_province_modifier = highway_robber_band
				has_province_modifier = incompetent_rule
				has_province_modifier = province_poverty_1
				has_province_modifier = province_poverty_2
			}
			any_province_character = { # a province character is ill or is a scholar and must not already be a prophet
				OR = {
					trait = infirm
					trait = ill
					trait = possessed
					trait = wounded
					trait = scholar
				}
				NOT = { has_character_modifier = prophet }
			}
		}
	}
	mean_time_to_happen = {
		days = 20
	}
	immediate = {
		random_realm_province = {
			limit = { # limit random province to the same conditions as the trigger
				OR = {
					has_province_modifier = thieves_guild
					has_province_modifier = smugglers_ring
					has_province_modifier = highway_robber_band
					has_province_modifier = incompetent_rule
					has_province_modifier = province_poverty_1
					has_province_modifier = province_poverty_2
				}
				any_province_character = {
					OR = {
						trait = infirm
						trait = ill
						trait = possessed
						trait = wounded
						trait = scholar
					}
					NOT = { has_character_modifier = prophet }
				}
			}
			random_province_character = {
				limit = { # limit random character to the same conditions as the trigger
					OR = {
						trait = infirm
						trait = ill
						trait = possessed
						trait = wounded
						trait = scholar
					}
					NOT = { has_character_modifier = prophet }
				}
				IF = { # If both a mono and poly is unfounded, choose random between groups
					limit = {
						NOT = { has_global_flag = polytheism_founded }
						NOT = { has_global_flag = monotheism_founded }
					}
					random_list = {
						50 = {
							character_event = { id = 1000190101 }
						}
						50 = {
							character_event = { id = 1000190401 }
						}
					}
				}
				IF = { # if only poly is unfounded, choose that
					limit = {
						NOT = { has_global_flag = polytheism_founded }
						has_global_flag = monotheism_founded
					}
					character_event = { id = 1000190101 }
				}
				IF = { # if only mono is unfounded, choose that
					limit = {
						has_global_flag = polytheism_founded
						NOT = { has_global_flag = monotheism_founded }
					}
					character_event = { id = 1000190401 }
				}
			}
		}
	}
}
